import java.util.Scanner;
import org.apache.commons.net.util.SubnetUtils;

public class SubnetCalculator {
public static void main(String[] args) {
Scanner scanner = new Scanner(System.in);

// Prompt user for IP address
System.out.print("Enter IP address (e.g., 192.168.1.0): ");
String ipAddress = scanner.nextLine();

// Prompt user for subnet mask length
System.out.print("Enter subnet mask length (e.g., 24 for /24 subnet): ");
int subnetMaskLength = scanner.nextInt();

scanner.close();

// Create SubnetUtils instance
SubnetUtils subnetUtils = new SubnetUtils(ipAddress + "/" + subnetMaskLength);
subnetUtils.setInclusiveHostCount(true); // Include network and broadcast addresses

// Get subnet information
String[] subnets = subnetUtils.getInfo().getAllAddresses();

// Print subnet information
for (String subnet : subnets) {
System.out.println("Subnet Address: " + subnet);
}
}
}
